📚 СИНТАКСИС POSTGRESQL (DDL, DML, DQL, DCL, индексы, функции, окна, CTE)

1. DDL (Data Definition Language) — создание и изменение структуры БД

-- Создание базы данных CREATE DATABASE mydb WITH ENCODING='UTF8' OWNER=postgres; -- Удаление базы данных DROP DATABASE IF EXISTS mydb; -- Создание схемы CREATE SCHEMA IF NOT EXISTS myschema; -- Создание таблицы (все типы данных) CREATE TABLE employees ( id SERIAL PRIMARY KEY, -- автоинкремент name VARCHAR(100) NOT NULL, email TEXT UNIQUE, age INT CHECK (age >= 18), salary NUMERIC(10,2) DEFAULT 0.0, is_active BOOLEAN DEFAULT true, created_at TIMESTAMP DEFAULT NOW(), profile JSONB, -- JSON тип tags TEXT[] -- массив ); -- Изменение таблицы ALTER TABLE employees ADD COLUMN department_id INT; ALTER TABLE employees RENAME COLUMN name TO full_name; ALTER TABLE employees ALTER COLUMN salary SET NOT NULL; ALTER TABLE employees DROP COLUMN tags; -- Создание внешнего ключа ALTER TABLE employees ADD CONSTRAINT fk_dept FOREIGN KEY (department_id) REFERENCES departments(id) ON DELETE CASCADE; -- Создание составного индекса CREATE INDEX idx_emp_name_age ON employees(full_name, age); -- Создание уникального индекса CREATE UNIQUE INDEX idx_unique_email ON employees(email); -- Удаление индекса DROP INDEX idx_emp_name_age; -- Создание представления (VIEW) CREATE VIEW active_employees AS SELECT id, full_name, salary FROM employees WHERE is_active = true; -- Материализованное представление (кешируется) CREATE MATERIALIZED VIEW emp_summary AS SELECT department_id, COUNT(*) FROM employees GROUP BY department_id; REFRESH MATERIALIZED VIEW emp_summary; -- Последовательности (SEQUENCE) CREATE SEQUENCE order_seq START 1000 INCREMENT 1; SELECT nextval('order_seq');

2. DML (Data Manipulation Language) — вставка, обновление, удаление

-- Вставка одной строки INSERT INTO employees (full_name, email, age, salary) VALUES ('Иван Петров', 'ivan@mail.ru', 25, 50000); -- Вставка нескольких строк INSERT INTO employees (full_name, email, age) VALUES ('Анна Сидорова', 'anna@mail.ru', 30), ('Олег Иванов', 'oleg@mail.ru', 28); -- Вставка из SELECT INSERT INTO employees_archive SELECT * FROM employees WHERE age > 50; -- UPDATE с условием UPDATE employees SET salary = salary * 1.1 WHERE department_id = 5; -- UPDATE с подзапросом UPDATE employees SET salary = (SELECT AVG(salary) FROM employees WHERE department_id = 1) WHERE id = 100; -- DELETE с условием DELETE FROM employees WHERE is_active = false; -- TRUNCATE (быстрое удаление всех строк) TRUNCATE TABLE temp_data;

3. DQL (Data Query Language) — SELECT любой сложности

-- Простая выборка SELECT id, full_name, salary FROM employees WHERE age > 25 ORDER BY salary DESC LIMIT 10; -- DISTINCT (уникальные значения) SELECT DISTINCT department_id FROM employees; -- Агрегатные функции (SUM, AVG, COUNT, MIN, MAX) SELECT department_id, AVG(salary) as avg_salary, COUNT(*) as emp_count FROM employees GROUP BY department_id HAVING COUNT(*) > 5; -- JOIN (INNER, LEFT, RIGHT, FULL) SELECT e.full_name, d.name as dept_name FROM employees e INNER JOIN departments d ON e.department_id = d.id; -- LEFT JOIN с проверкой на NULL SELECT e.full_name, d.name FROM employees e LEFT JOIN departments d ON e.department_id = d.id; -- Подзапросы (скалярные, табличные, EXISTS) SELECT full_name FROM employees WHERE salary > (SELECT AVG(salary) FROM employees); -- EXISTS (проверка наличия) SELECT d.name FROM departments d WHERE EXISTS (SELECT 1 FROM employees e WHERE e.department_id = d.id); -- CTE (Common Table Expressions — WITH) WITH high_salary AS ( SELECT * FROM employees WHERE salary > 70000 ) SELECT * FROM high_salary WHERE age < 40; -- Рекурсивный CTE (иерархия) WITH RECURSIVE org_tree AS ( SELECT id, name, manager_id, 1 AS level FROM employees WHERE manager_id IS NULL UNION ALL SELECT e.id, e.name, e.manager_id, ot.level + 1 FROM employees e INNER JOIN org_tree ot ON e.manager_id = ot.id ) SELECT * FROM org_tree; -- Оконные функции (WINDOW FUNCTIONS) SELECT full_name, salary, department_id, ROW_NUMBER() OVER (PARTITION BY department_id ORDER BY salary DESC) as rank_in_dept, AVG(salary) OVER (PARTITION BY department_id) as dept_avg, LAG(salary) OVER (ORDER BY id) as prev_salary FROM employees; -- FULLTEXT поиск (tsvector/tsquery) CREATE INDEX idx_fts ON documents USING GIN(to_tsvector('russian', content)); SELECT * FROM documents WHERE to_tsvector('russian', content) @@ to_tsquery('russian', 'AI & базы'); -- Работа с JSONB SELECT profile->>'city' as city FROM employees WHERE profile @> '{"experience": "senior"}';

7. Векторные операции pgvector (расширение для AI)

CREATE EXTENSION vector; CREATE TABLE products ( id SERIAL, name TEXT, embedding VECTOR(384) -- размерность эмбеддинга ); -- Вставка векторов INSERT INTO products VALUES (1, 'Ноутбук', '[0.123, -0.456, 0.789, ...]'); -- Операторы векторного поиска SELECT * FROM products ORDER BY embedding <-> '[0.1,0.2,0.3]' LIMIT 5; -- L2 SELECT * FROM products ORDER BY embedding <#> '[0.1,0.2,0.3]' LIMIT 5; -- косинус SELECT * FROM products ORDER BY embedding <#> '[0.1,0.2,0.3]' LIMIT 5; -- скалярное -- Индексы для векторов CREATE INDEX ON products USING ivfflat (embedding vector_cosine_ops) WITH (lists = 100); CREATE INDEX ON products USING hnsw (embedding vector_cosine_ops);

📐 ХАРАКТЕРИСТИКА ВЕКТОРНЫХ ЭМБЕДДИНГОВ

Что такое эмбеддинг (embedding)?
Эмбеддинг — это числовое представление объекта (текста, изображения, аудио, пользователя, товара) в виде вектора фиксированной размерности. Нейросеть обучается преобразовывать объекты так, чтобы семантически похожие объекты имели близкие векторы в многомерном пространстве.

Ключевые свойства эмбеддингов:

  • Размерность (dimensionality) — количество чисел в векторе. Типичные значения: 128, 256, 384, 768, 1024, 1536. Чем выше размерность, тем больше информации может быть закодировано, но тем больше памяти требуется.
  • Нормализация — часто векторы нормализуют (длина = 1), чтобы косинусное расстояние совпадало со скалярным произведением.
  • Плотность — эмбеддинги плотные (почти все значения ненулевые), в отличие от разреженных one-hot векторов.
  • Семантическое пространство — координаты не интерпретируются напрямую, важны расстояния между векторами.

Как получают эмбеддинги? (модели и подходы)

# Python пример получения эмбеддинга текста через sentence-transformers from sentence_transformers import SentenceTransformer # Загружаем модель (размерность 384) model = SentenceTransformer('all-MiniLM-L6-v2') # Получаем эмбеддинг для текста text = "Искусственный интеллект в базах данных" embedding = model.encode(text) # массив из 384 чисел print(embedding.shape) # (384,) print(embedding[:5]) # [-0.123, 0.456, -0.789, 0.234, -0.567]

Математические операции с эмбеддингами:

-- Косинусное расстояние: 1 - (A·B)/(|A|*|B|) -- Чем меньше, тем ближе по смыслу cosine_distance = 1 - (dot_product / (norm(a) * norm(b))) -- Евклидово (L2) расстояние: sqrt(Σ(a_i - b_i)²) -- Хорошо для координатных данных -- Скалярное произведение: Σ(a_i * b_i) -- Для нормализованных векторов совпадает с косинусом

Примеры использования эмбеддингов в реальных задачах:

  • Поиск дубликатов: найти документы с эмбеддингом, близким к данному.
  • Кластеризация: сгруппировать похожие объекты (k-means, DBSCAN).
  • Рекомендательные системы: "пользователи, похожие на вас" или "товары, похожие на этот".
  • RAG (Retrieval-Augmented Generation): найти релевантные документы для LLM.
  • Поиск аномалий: объекты, чьи эмбеддинги далеки от всех остальных.

Важные практические советы:

  • Всегда нормализуйте эмбеддинги, если используете косинусное расстояние.
  • Размерность эмбеддинга должна быть одинаковой для всех записей в таблице.
  • Для больших наборов данных (более 100 000 векторов) обязательно создавайте индексы IVFFlat или HNSW.
  • Эмбеддинги можно агрегировать (усреднять) для получения вектора группы объектов.
  • Перед вставкой в БД эмбеддинги можно сжать (квантование, PCA) для экономии места.
📌 Зачем нужны AI-ориентированные СУБД? Проблема традиционных решений

Обычные реляционные БД (PostgreSQL, MySQL, Oracle) спроектированы для точного совпадения и сравнения по порядку. Они используют индексы B-Tree или Hash для операторов =, >, LIKE (с ограничениями). Однако они не могут ответить на запросы типа: "найди 10 товаров, похожих на этот" или "найди документы, семантически близкие к тексту запроса". Классический SQL не имеет встроенных средств для измерения близости в многомерном пространстве.

❌ Пример НЕПРАВИЛЬНОГО подхода (жесткие фильтры вместо семантики):

-- Так НЕ работает для семантического поиска: SELECT * FROM products WHERE category = 'электроника' AND price BETWEEN 500 AND 1000; -- Это жесткие фильтры, а не поиск похожих объектов! -- Такой запрос НЕ найдет "похожий" товар, если у него другая категория или цена выходит за диапазон.

✅ Правильный подход с использованием pgvector (векторный поиск):

-- 1. Создаем таблицу с векторной колонкой (эмбеддинги товаров) CREATE EXTENSION vector; CREATE TABLE products ( id SERIAL PRIMARY KEY, name TEXT, category TEXT, price NUMERIC, embedding VECTOR(384) -- эмбеддинг, описывающий товар семантически ); -- 2. Добавляем индекс для быстрого поиска по косинусному расстоянию CREATE INDEX ON products USING ivfflat (embedding vector_cosine_ops) WITH (lists = 100); -- 3. Поиск 10 товаров, семантически похожих на заданный (по эмбеддингу) -- Вместо жестких условий по категории и цене — измеряем косинусную близость! SELECT name, category, price, 1 - (embedding <#> '[0.123, -0.456, 0.789, ...]') AS similarity FROM products ORDER BY embedding <#> '[0.123, -0.456, 0.789, ...]' LIMIT 10; -- 4. Комбинированный поиск: фильтр + семантическая близость SELECT name, price, 1 - (embedding <#> target_embedding) AS similarity FROM products WHERE category = 'электроника' -- можно добавить традиционный фильтр AND price BETWEEN 500 AND 1000 ORDER BY embedding <#> target_embedding LIMIT 10;

Объяснение примера с pgvector:

  • Векторная колонка embedding хранит числовое представление товара (нейросеть превращает название, описание, категорию в вектор).
  • Индекс ivfflat ускоряет поиск ближайших соседей (ANN) в сотни раз.
  • Оператор <#> вычисляет косинусное расстояние — чем оно меньше, тем семантически ближе товары.
  • В отличие от жестких фильтров, векторный поиск найдет товары, которые похожи по смыслу, даже если у них разные категории или цены.
  • Можно комбинировать традиционные фильтры (WHERE category = ...) с векторным поиском для гибридных запросов.

Таким образом, pgvector превращает PostgreSQL в полноценную AI-ориентированную СУБД, сохраняя все преимущества реляционной модели.

🏗️ Архитектура AI-ориентированной СУБД на примере pgvector

pgvector — это расширение PostgreSQL, которое добавляет:

  • Тип данных VECTOR(n) — хранит массив чисел с плавающей точкой фиксированной размерности.
  • Операторы векторного поиска — <-> (L2), <#> (косинус), <#> (скалярное произведение).
  • Индексы ANN — IVFFlat и HNSW, ускоряющие поиск в сотни раз.
  • Полную интеграцию с SQL — все стандартные конструкции (WHERE, JOIN, ORDER BY) работают с векторами.

Благодаря тому, что pgvector является расширением, а не отдельной БД, вы получаете ACID-транзакции, репликацию, резервное копирование и все богатство экосистемы PostgreSQL.

Рекомендации по выбору:

  • pgvector — лучший выбор для production веб-приложений, если уже используется PostgreSQL. Дает баланс между функциональностью, производительностью и удобством.

🧪 Как получаются векторы? Эмбеддинги от нейросетей (практический аспект)

Сырые данные (текст, изображение, аудио) нельзя напрямую загрузить в векторную БД. Их нужно пропустить через нейросеть-энкодер. Популярные модели:

  • Sentence Transformers (SBERT) — модель all-MiniLM-L6-v2 даёт векторы размерности 384, отлично подходит для русского/английского текста.
  • OpenAI Ada-002 — проприетарные эмбеддинги размерности 1536, высокое качество.
  • CLIP — объединяет текст и изображения в общем пространстве, позволяет искать картинки по тексту и наоборот.
  • RuBERT — русскоязычные эмбеддинги от DeepPavlov / SberDevices.
# Python пример получения эмбеддинга через sentence-transformers from sentence_transformers import SentenceTransformer model = SentenceTransformer('all-MiniLM-L6-v2') embedding = model.encode("AI системы управления базами данных") # embedding теперь вектор размерности 384, можно сохранить в pgvector print(embedding[:5]) # первые 5 чисел: [0.123, -0.456, 0.789, ...]

На практических занятиях мы используем ручное кодирование (цвета, жанры), чтобы понять принцип, а затем можно перейти к реальным эмбеддингам через Python/JavaScript.

🚀 Производительность: почему индексы важны

Без индекса PostgreSQL выполняет последовательное сканирование (Seq Scan) — O(N) сравнений. При 1 млн векторов поиск занимает секунды. Индексы используют алгоритмы приближенного поиска ближайших соседей (ANN), которые жертвуют небольшой точностью ради огромного ускорения.

-- План запроса БЕЗ индекса: Limit (cost=0.00..1234.56 rows=10 width=100) -> Sort (cost=0.00..12345.67 rows=100000 width=100) Sort Key: ((vec <#> '[0.5,0.5,...]')) -> Seq Scan on vectors (cost=0.00..1234.56 rows=100000) Время выполнения: ~180 ms -- План запроса С индексом HNSW: Limit (cost=0.00..12.34 rows=10 width=100) -> Index Scan using hnsw_idx on vectors Order By: (vec <#> '[0.5,0.5,...]') Время выполнения: ~3 ms

Ускорение в 60 раз! Поэтому для реальных приложений создание индексов — обязательный этап.

3 Векторы, эмбеддинги и операторы поиска
📐 Понятие вектора и эмбеддинга: Вектор — упорядоченный список чисел. Эмбеддинг — векторное представление объекта (текста, изображения, товара), полученное нейросетью. Размерность определяет детализацию. Пример: жанр «боевик» → [0.9, 0.1, 0.2]. Близость векторов = семантическая похожесть.
🎨 Практическая работа №2 — Ручное создание простых векторов (цвета, жанры)
  1. CREATE TABLE colors (name TEXT, rgb VECTOR(3));
  2. INSERT INTO colors VALUES ('Красный','[1,0,0]'),('Зеленый','[0,1,0]'),('Синий','[0,0,1]'),('Желтый','[1,1,0]');
  3. SELECT name, rgb <-> '[1,0.5,0]' AS dist FROM colors ORDER BY dist LIMIT 1;
CREATE TABLE colors (name TEXT, rgb VECTOR(3));
INSERT INTO colors VALUES ('Красный','[1,0,0]'),('Зеленый','[0,1,0]'),('Синий','[0,0,1]'),('Желтый','[1,1,0]');
SELECT name, rgb <-> '[1,0.5,0]' AS euclidean_distance FROM colors ORDER BY euclidean_distance LIMIT 1; -- Желтый
📏 Операторы векторного поиска:
• <-> — Евклидово (L2)
• <#> — Косинусное расстояние
• <#> — Скалярное произведение
🧪 Практическая работа №3 — Написание запросов с разными операторами pgvector

  2. Найдите ближайшую точку к (1.1,1.1) с L2 и косинусом.
